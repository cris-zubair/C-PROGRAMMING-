#include<stdio.h>
int main(){
    int i,min=0,a[5];
    for(i=0;i<5;i++){
        scanf("%d",&a[i]);
    }
    min=a[0];
    for(i=0;i<5;i++){
        if(min>a[i])
        min=a[i];
    }
    printf("min=%d",min);
}

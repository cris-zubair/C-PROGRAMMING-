#include<stdio.h>
int main(){
    int i,max=0,a[5];
    for(i=0;i<5;i++){
        scanf("%d",&a[i]);
    }
    max=a[0];
    for(i=0;i<5;i++){
        if(max<a[i])
        max=a[i];
    }
    printf("max=%d",max);
}

WHILE LOOP PROGRAM
---------------------------------------------------------------------
#include<stdio.h>
int main(){
    int i=1,n,r;
    printf("Enter a number:");
    scanf("%d",&n);
    printf("Enter the range: ");
    scanf("%d",&r);
    while(i<=r){
        printf("%d * %d =%d \n",n,i,n*i);
        i++;
    }
    return 0;
}
